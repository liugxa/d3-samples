Usage Example
var smtTime = new Date("2015-12-31T00:00:00.0000Z");
var stTime = new Date("2015-12-31T00:10:00.0000Z");
var ctTime = new Date("2015-12-31T00:20:00.0000Z");
var eetTime = new Date("2015-12-31T00:59:00.0000Z");
var rtlTime = null;

//initialize the context
var container = d3.select("#div0");
var context = new ProgressBarContext(container);

//initilize the progress bar object
var progressBar = new RunningProgressBar(context, smtTime, stTime, ctTime, eetTime, rtlTime).init();

//show all of the progressers
var progressers = progressBar.getProgressers();
for(var i=0;i < progressers.length;i++) progressers[i].show();

//show all of the timers
var timers = progressBar.getTimers();
for(var i=0;i < timers.length;i++) timers[i].show();
